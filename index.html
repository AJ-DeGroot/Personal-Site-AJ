class Tree
  attr_accessor :payload, :children

  def initialize(payload, children)
    @payload = payload
    @children = children
  end

  def search_depth(branch)
    @stack = []
    puts 'Depth Search:'
    depth(branch)
    puts '-------------'
  end

  def search_breadth(branch)
    @queue = []
    @queue << branch
    puts 'Breadth Search:'
    breadth(branch)
    puts '-------------'
  end
  
  def depth(branch)
    if check_value(branch)
    elsif branch.children.count != 0
      until branch.children.empty?
        @stack << branch.children.pop
      end
      depth(@stack.last)
    else
      @stack.pop
      depth(@stack.last)
    end
  end

  def breadth(branch)
    until @queue.count == 0 || check_value(@queue.first)
      until @queue.first.children.count == 0
        @queue << @queue.first.children.shift
      end
      @queue.shift
    end
  end

  def check_value(branch)
    puts branch.payload 
    if branch.payload == 11
      return branch.payload
    end
  end
end

Temperature Conversion

This Javascript program will prompt the user for either Clesius or Fahrenheit the user know what the other temperature is. 

var readlineSync = require('readline-sync'),
  conversions = ['Celcius to Fahrenheit', 'Fahrenheit to Celcius'],
  index = readlineSync.keyInSelect(conversions, 'Which Conversion?');

if (index == 0) {
  console.log('')
  var degrees = readlineSync.question('Enter degrees in Celsius: ');
  var degreesNum = Number(degrees);
  var degreesFahrenheit = degreesNum * 1.8 + 32;
  console.log('It is ' + degreesFahrenheit + ' degrees Fahrenheit');
  console.log('')
} else if (index==1) {
  console.log('')
  var degrees = readlineSync.question('Enter degrees in Fahrenheit: ');
  var degreesNum = Number(degrees);
  var degreesCelcius = (degreesNum - 32) / 1.8;
  console.log('It is ' + degreesCelcius + ' degrees Celcius');
  console.log('')
} else {
  console.log('Exit')
}

Foobar

This ruby program is an adaptation of a classic technical interview problem which displays sequences of the Foobar pattern. 

puts "What number will we be counting up to today?:"
user_number = gets.chomp.to_i
puts "----"

n = 1

while n <= user_number
    if n % 5 == 0 && n % 3 == 0
        puts "Foobar"
    elsif n % 3 == 0
        puts "Foo"
    elsif n % 5 == 0
        puts "Bar"
    else
        puts n
    end
    
    n = n + 1 
end

def to_i(input)
  input = input.split('.')[0] if input.include?('.')
  sign = input[0] == '-' ? -1 : 1
  input.gsub!(/\A-/, '')

  total = 0
  input.chars.reverse.each_with_index do |num, index|
    total = total + ((num.ord - 48) * 10 ** index)
  end

  total * sign
end
